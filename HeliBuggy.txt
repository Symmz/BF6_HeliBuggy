// ===== MATCH START CONFIGURATION =====
let baseGameStartCountdown: number = 15; // Countdown duration in seconds
let gameStartCountdown: number = 15;
let gameStartDebounce: boolean = false; // Prevents countdown from restarting once game has started

// ===== CUSTOM MATCH TIMER =====
let customMatchTimeRemaining: number = 300; // 5 minutes in seconds
let customTimerStarted: boolean = false;

// ===== HELPER FUNCTIONS =====
function FormatTime(sec: number): [number, number, number] {
    const minutes = Math.floor(sec / 60);
    const seconds = Math.floor(sec % 60);
    const coolerSeconds = Math.floor((sec % 60) / 10);
    return [minutes, coolerSeconds, seconds];
}

function RestrictPlayerMovement(player: mod.Player) {
    // Restrict all player input during match waiting
    mod.EnableAllInputRestrictions(player, true);
}

function MakeMessage(message: string, ...args: any[]) {
    switch (args.length) {
        case 0:
            return mod.Message(message);
        case 1:
            return mod.Message(message, args[0]);
        case 2:
            return mod.Message(message, args[0], args[1]);
        case 3:
            return mod.Message(message, args[0], args[1], args[2]);
        default:
            throw new Error("Invalid number of arguments");
    }
}

// Helper function to hide the timer UI
function HideMatchTimer() {
    const timerWidget = mod.FindUIWidgetWithName("RemainingTime");
    mod.SetUIWidgetVisible(timerWidget, false);
    console.log("Timer widget hidden");
}

// Helper function to format custom timer
function FormatCustomTimer(seconds: number): mod.Message {
    const mins = Math.floor(seconds / 60);
    const secs = Math.floor(seconds % 60);
    const secsTens = Math.floor(secs / 10);
    const secsOnes = secs % 10;
    return mod.Message("{0}:{1}{2}", mins, secsTens, secsOnes);
}

// ===== PLAYER PROFILE CLASS =====
class PlayerProfile {
    player: mod.Player;
    playerWaitingHeaderWidget: mod.UIWidget;
    playerWaitingSubHeaderWidget: mod.UIWidget;
    playerCountdownWidget: mod.UIWidget;
    
    static playerInstances: mod.Player[] = [];
    static deployedPlayers: mod.Player[] = [];
    static whileLoopStarted: boolean = false;
    static #allPlayerProfiles: { [key: number]: PlayerProfile } = {};

    constructor(player: mod.Player) {
        console.log("Creating PlayerProfile for player: " + mod.GetObjId(player));
        this.player = player;
        this.playerWaitingHeaderWidget = this.CreateWaitingHeaderUI();
        this.playerWaitingSubHeaderWidget = this.CreateWaitingSubHeaderUI();
        this.playerCountdownWidget = this.CreateCountdownUI();
        PlayerProfile.playerInstances.push(this.player);
        console.log("PlayerProfile created. Total instances: " + PlayerProfile.playerInstances.length);
    }

    static Get(player: mod.Player): PlayerProfile | undefined {
        if (mod.GetObjId(player) > -1) {
            let index = mod.GetObjId(player);
            let playerProfile = this.#allPlayerProfiles[index];

            if (!playerProfile) {
                console.log("PlayerProfile not found for player " + index + ", creating new one");
                playerProfile = new PlayerProfile(player);
                this.#allPlayerProfiles[index] = playerProfile;
            } else {
                console.log("PlayerProfile found for player " + index);
            }
            return playerProfile;
        }
        console.log("Invalid player ID, returning undefined");
        return undefined;
    }

    static RemovePlayerFromDeployedPlayers(player: mod.Player) {
        const index = this.deployedPlayers.findIndex(
            (p) => mod.GetObjId(p) === mod.GetObjId(player)
        );
        if (index !== -1) {
            this.deployedPlayers.splice(index, 1);
        }
    }

    // ===== CORE COUNTDOWN LOGIC =====
    static async UpdateGameStartCountdown() {
        console.log("UpdateGameStartCountdown called. gameStartDebounce: " + gameStartDebounce + ", whileLoopStarted: " + this.whileLoopStarted);
        
        if (gameStartDebounce || this.whileLoopStarted) {
            console.log("Returning early from UpdateGameStartCountdown");
            return;
        }

        // Count DEPLOYED (alive) players on Team 1 and Team 2
        const allPlayers = mod.AllPlayers();
        let team1Count = 0;
        let team2Count = 0;
        
        for (let i = 0; i < mod.CountOf(allPlayers); i++) {
            const player = mod.ValueInArray(allPlayers, i);
            // Only count non-AI players who are ALIVE (deployed)
            if (mod.GetSoldierState(player, mod.SoldierStateBool.IsAlive)) {
                const playerTeam = mod.GetTeam(player);
                const team1 = mod.GetTeam(1);
                const team2 = mod.GetTeam(2);
                
                if (mod.Equals(playerTeam, team1)) {
                    team1Count++;
                } else if (mod.Equals(playerTeam, team2)) {
                    team2Count++;
                }
            }
        }

        console.log("Team 1 DEPLOYED players: " + team1Count + ", Team 2 DEPLOYED players: " + team2Count);

        // Start countdown only if at least 1 DEPLOYED player is on each team
        if (
            team1Count >= 1 &&
            team2Count >= 1 &&
            !this.whileLoopStarted
        ) {
            console.log("Both teams have deployed players, starting countdown");
            this.whileLoopStarted = true;
            
            // Hide the "Waiting for Players" text before countdown starts
            for (let i = 0; i < this.playerInstances.length; i++) {
                const element = this.playerInstances[i];
                const playerProfile = this.Get(element);
                if (playerProfile) {
                    playerProfile.UpdateUI(playerProfile.playerWaitingHeaderWidget, undefined, false);
                    playerProfile.UpdateUI(playerProfile.playerWaitingSubHeaderWidget, undefined, false);
                }
            }
            
            // Small delay to ensure UI clears
            await mod.Wait(0.5);
            
            while (true) {
                // Recount DEPLOYED players on each team every iteration
                const currentPlayers = mod.AllPlayers();
                let currentTeam1Count = 0;
                let currentTeam2Count = 0;
                
                for (let i = 0; i < mod.CountOf(currentPlayers); i++) {
                    const player = mod.ValueInArray(currentPlayers, i);
                    // Only count non-AI players who are ALIVE (deployed)
                    if (mod.GetSoldierState(player, mod.SoldierStateBool.IsAlive)) {
                        const playerTeam = mod.GetTeam(player);
                        const team1 = mod.GetTeam(1);
                        const team2 = mod.GetTeam(2);
                        
                        if (mod.Equals(playerTeam, team1)) {
                            currentTeam1Count++;
                        } else if (mod.Equals(playerTeam, team2)) {
                            currentTeam2Count++;
                        }
                    }
                }
                
                // Update countdown display for all player instances
                for (let i = 0; i < this.playerInstances.length; i++) {
                    const element = this.playerInstances[i];
                    const playerProfile = this.Get(element);
                    if (playerProfile) {
                        // FIXED: Display countdown as simple number (15, 14, 13, etc.)
                        playerProfile.UpdateUI(
                            playerProfile.playerCountdownWidget,
                            mod.Message("{0}", gameStartCountdown),
                            true
                        );
                    }
                    
                    // Release restrictions when countdown hits 0
                    if (
                        gameStartCountdown <= 0 &&
                        mod.GetSoldierState(element, mod.SoldierStateBool.IsAlive)
                    ) {
                        mod.EnableAllInputRestrictions(element, false);
                    }
                }

                // Reset countdown if either team has 0 DEPLOYED players
                if (currentTeam1Count < 1 || currentTeam2Count < 1) {
                    console.log("Not enough deployed players on both teams (Team 1: " + currentTeam1Count + ", Team 2: " + currentTeam2Count + "), resetting countdown");
                    gameStartCountdown = baseGameStartCountdown;
                    this.whileLoopStarted = false;
                    
                    // Hide countdown widget and show waiting UI again
                    for (let i = 0; i < this.playerInstances.length; i++) {
                        const element = this.playerInstances[i];
                        const playerProfile = this.Get(element);
                        if (playerProfile) {
                            playerProfile.UpdateUI(playerProfile.playerCountdownWidget, undefined, false);
                        }
                    }
                    break;
                }

                // Countdown complete - start the game!
                if (gameStartCountdown <= 0 && !gameStartDebounce) {
                    console.log("Countdown complete! Starting game...");
                    gameStartDebounce = true;
                    
                    // FIXED: Set game as ongoing and START custom 5-minute timer
                    mod.SetVariable(GameOngoingGlobalVar, true);
                    customMatchTimeRemaining = 300; // Reset to 5 minutes
                    customTimerStarted = true; // Start the custom timer
                    console.log("Custom 5-minute timer started!");
                    
                    // Make timer visible
                    const timerWidget = mod.FindUIWidgetWithName("RemainingTime");
                    mod.SetUIWidgetVisible(timerWidget, true);
                    console.log("Timer widget made visible");
                    
                    // Hide countdown widget
                    for (let i = 0; i < this.playerInstances.length; i++) {
                        const element = this.playerInstances[i];
                        const playerProfile = this.Get(element);
                        if (playerProfile) {
                            playerProfile.UpdateUI(playerProfile.playerCountdownWidget, undefined, false);
                        }
                    }
                    
                    // Show GO message briefly
                    await mod.Wait(0.5);
                    for (let i = 0; i < this.playerInstances.length; i++) {
                        const element = this.playerInstances[i];
                        const playerProfile = this.Get(element);
                        if (playerProfile) {
                            playerProfile.UpdateUI(
                                playerProfile.playerWaitingHeaderWidget,
                                MakeMessage("GO!"),
                                true
                            );
                            playerProfile.UpdateUI(
                                playerProfile.playerWaitingSubHeaderWidget,
                                MakeMessage("Team 1 must reach the capture point!"),
                                true
                            );
                        }
                    }
                    
                    await mod.Wait(3);
                    
                    // Hide GO message after game starts
                    for (let i = 0; i < this.playerInstances.length; i++) {
                        const element = this.playerInstances[i];
                        const playerProfile = this.Get(element);
                        if (playerProfile) {
                            playerProfile.UpdateUI(playerProfile.playerWaitingHeaderWidget, undefined, false);
                            playerProfile.UpdateUI(playerProfile.playerWaitingSubHeaderWidget, undefined, false);
                        }
                    }
                    
                    console.log("Match started! Custom timer should now be counting down.");
                    break;
                }

                await mod.Wait(1);
                gameStartCountdown--;
            }
        } else {
            // Not enough deployed players on both teams - show waiting message
            gameStartCountdown = baseGameStartCountdown;
        }
    }

        mod.SetUIWidgetBgColor(widget, mod.CreateVector(0, 0, 0));

        mod.SetUITextSize(widget, 72);  // Much larger text for countdown
        mod.SetUITextColor(widget, mod.CreateVector(1, 0.8, 0));  // Gold/yellow color
        mod.SetUIWidgetBgColor(widget, mod.CreateVector(0, 0, 0));
        mod.SetUIWidgetBgAlpha(widget, 0.8);
        mod.SetUIWidgetBgFill(widget, mod.UIBgFill.Blur);
        mod.SetUIWidgetDepth(widget, mod.UIDepth.AboveGameUI);
        mod.SetUIWidgetVisible(widget, false);  // Start hidden
        
        return widget;
    }

    UpdateUI(widget: mod.UIWidget, message: mod.Message | undefined, visible: boolean) {
        if (message !== undefined) {
            mod.SetUITextLabel(widget, message);
        }
        mod.SetUIWidgetVisible(widget, visible);
    }

    ShowLoadoutWarningUI(show: boolean) {
        // Placeholder for loadout warning functionality
    }
}

   }
 }

// Modified CheckAndUpdateWaitingState to check for Team 1 and Team 2 specifically
function CheckAndUpdateWaitingState() {
  console.log("CheckAndUpdateWaitingState called");
  
  // Get all current players and count DEPLOYED (alive) players by team
  const allPlayers = mod.AllPlayers();
  let team1Count = 0;
  let team2Count = 0;
  
  for (let i = 0; i < mod.CountOf(allPlayers); i++) {
    const player = mod.ValueInArray(allPlayers, i);
    // Only count non-AI players who are ALIVE (deployed)
    if (mod.GetSoldierState(player, mod.SoldierStateBool.IsAlive)) {
      const playerTeam = mod.GetTeam(player);
      const team1 = mod.GetTeam(1);
      const team2 = mod.GetTeam(2);
      
      if (mod.Equals(playerTeam, team1)) {
        team1Count++;
      } else if (mod.Equals(playerTeam, team2)) {
        team2Count++;
      }
    }
  }
  
  console.log("Team 1 DEPLOYED players: " + team1Count + ", Team 2 DEPLOYED players: " + team2Count);
  
  // Only show waiting UI if countdown hasn't started yet
  if (!PlayerProfile.whileLoopStarted && !gameStartDebounce) {
    // Ensure all players have profiles and show them waiting UI
    for (let i = 0; i < mod.CountOf(allPlayers); i++) {
      const player = mod.ValueInArray(allPlayers, i);
      
      // Create profile if needed
      const playerProfile = PlayerProfile.Get(player);
      
      // Only show UI to alive players
      if (mod.GetSoldierState(player, mod.SoldierStateBool.IsAlive)) {
        console.log("Showing waiting UI to player: " + mod.GetObjId(player));
        
        if (playerProfile) {
          playerProfile.UpdateUI(
            playerProfile.playerWaitingHeaderWidget,
            MakeMessage("Waiting for Players..."),
            true
          );
          playerProfile.UpdateUI(
            playerProfile.playerWaitingSubHeaderWidget,
            MakeMessage(
              "Team 1: {} | Team 2: {}",
              team1Count,
              team2Count
            ),
            true
          );
        }
      }
    }
  }
  
  // Trigger countdown if both teams have at least 1 DEPLOYED player
  if (team1Count >= 1 && team2Count >= 1 && !gameStartDebounce) {
    console.log("Both teams have deployed players, triggering countdown");
    PlayerProfile.UpdateGameStartCountdown();
  }
}

// Ongoing check during waiting phase
function OngoingGlobal_Monitor_Waiting_Phase_Condition(): boolean {
  const newState = mod.Not(mod.GetVariable(GameOngoingGlobalVar));
  return newState;
}

async function OngoingGlobal_Monitor_Waiting_Phase_Action() {
  // Continuously run during waiting phase
  while (true) {
    // Only check if game hasn't started
    if (!gameStartDebounce) {
      // Get all alive players and restrict their movement
      const allPlayers = mod.AllPlayers();
      for (let i = 0; i < mod.CountOf(allPlayers); i++) {
        const player = mod.ValueInArray(allPlayers, i);
        
        
        // Restrict movement for alive players during waiting phase
        if (mod.GetSoldierState(player, mod.SoldierStateBool.IsAlive)) {
          mod.EnableAllInputRestrictions(player, true);
        }
      }
      
      // Check and update waiting state
      CheckAndUpdateWaitingState();
    }
    
    await mod.Wait(0.5);
  }
}

function OngoingGlobal_Monitor_Waiting_Phase(conditionState: any) {
  let newState = OngoingGlobal_Monitor_Waiting_Phase_Condition();
  if (!conditionState.update(newState)) {
    return;
  }
  OngoingGlobal_Monitor_Waiting_Phase_Action();
}

// NEW: Check for team elimination during active match
function OngoingGlobal_Check_Team_Elimination_Condition(): boolean {
  // Only run when game is ongoing
  const newState = mod.GetVariable(GameOngoingGlobalVar);
  return newState;
}

async function OngoingGlobal_Check_Team_Elimination_Action() {
  // Continuously check every 3 seconds during the match
  while (true) {
    // Only check if game has started
    if (gameStartDebounce && mod.GetVariable(GameOngoingGlobalVar)) {
      // Count alive players on each team
      const allPlayers = mod.AllPlayers();
      let team1AliveCount = 0;
      let team2AliveCount = 0;
      
      for (let i = 0; i < mod.CountOf(allPlayers); i++) {
        const player = mod.ValueInArray(allPlayers, i);
        
        
        if (mod.GetSoldierState(player, mod.SoldierStateBool.IsAlive)) {
          const playerTeam = mod.GetTeam(player);
          const team1 = mod.GetTeam(1);
          const team2 = mod.GetTeam(2);
          
          if (mod.Equals(playerTeam, team1)) {
            team1AliveCount++;
          } else if (mod.Equals(playerTeam, team2)) {
            team2AliveCount++;
          }
        }
      }
      
      console.log("Team Elimination Check - Team 1 Alive: " + team1AliveCount + ", Team 2 Alive: " + team2AliveCount);
      
      // FIXED: Hide timer before ending game for team elimination
      // Check if either team has been eliminated
      if (team1AliveCount === 0 && team2AliveCount > 0) {
        console.log("Team 1 eliminated! Team 2 wins!");
        HideMatchTimer();
        mod.SetVariable(GameOngoingGlobalVar, false);
        customTimerStarted = false;
        mod.EndGameMode(mod.GetTeam(2));
        break;
      } else if (team2AliveCount === 0 && team1AliveCount > 0) {
        console.log("Team 2 eliminated! Team 1 wins!");
        HideMatchTimer();
        mod.SetVariable(GameOngoingGlobalVar, false);
        customTimerStarted = false;
        mod.EndGameMode(mod.GetTeam(1));
        break;
      }
    }
    
    await mod.Wait(3);
  }
}

function OngoingGlobal_Check_Team_Elimination(conditionState: any) {
  let newState = OngoingGlobal_Check_Team_Elimination_Condition();
  if (!conditionState.update(newState)) {
    return;
  }
  OngoingGlobal_Check_Team_Elimination_Action();
}

async function OnPlayerDeploy_Restrict_Movement_Action(eventInfo: any) {
  console.log("OnPlayerDeploy triggered for player: " + mod.GetObjId(eventInfo.eventPlayer));
  
  // Check if game has already started - if so, block all new deployments
  if (gameStartDebounce) {
    console.log("Game already started - BLOCKING deployment for player: " + mod.GetObjId(eventInfo.eventPlayer));
    // Force player back to deployment screen
    mod.EnablePlayerDeploy(eventInfo.eventPlayer, false);
    return;
  }
  
  // Ensure player profile is created
  PlayerProfile.Get(eventInfo.eventPlayer);
  
  // Wait a tiny bit to ensure player is fully deployed
  await mod.Wait(0.1);
  
  // Game hasn't started yet - restrict movement immediately and forcefully
  console.log("Restricting movement for player: " + mod.GetObjId(eventInfo.eventPlayer));
  mod.EnableAllInputRestrictions(eventInfo.eventPlayer, true);
  
  // Double-check restriction after a short delay to ensure it sticks
  await mod.Wait(0.2);
  mod.EnableAllInputRestrictions(eventInfo.eventPlayer, true);
  
  // Trigger the waiting state check to potentially start countdown
  CheckAndUpdateWaitingState();
}

function MakeTimeFromSeconds(seconds: number) {
    // Calculate minutes (divide by 60)
    let minutes = mod.Floor(mod.Divide(seconds, 60));
    
    // Calculate remaining seconds (modulo 60) and floor to remove decimals
    let remainingSeconds = mod.Floor(mod.Modulo(seconds, 60));
    
    // Split seconds into tens digit (0-5) and ones digit (0-9) like FormatTime does
    let secondsTens = mod.Floor(mod.Divide(remainingSeconds, 10));
    let secondsOnes = mod.Modulo(remainingSeconds, 10);
    
    // Return formatted string (e.g., "5:09" or "4:32")
    return mod.Message("{0}:{1}{2}", minutes, secondsTens, secondsOnes);
}
